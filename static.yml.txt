        function startParty() {
            // 1. Hide the button
            document.getElementById('welcome-screen').style.display = 'none';
            
            // 2. Show the content with flex display
            const content = document.getElementById('party-content');
            content.style.display = 'flex';

            // 3. Play the birthday audio (Optional - un-comment if you add music)
            // var audio = new Audio('happy-birthday.mp3');
            // audio.play();

            // 4. TRIGGER THE CONFETTI BOMB
            var duration = 3 * 1000;
            var end = Date.now() + duration;

            (function frame() {
                // Launch confetti from the left edge
                confetti({
                    particleCount: 5,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0 }
                });
                // Launch confetti from the right edge
                confetti({
                    particleCount: 5,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1 }
                });

                // Keep firing until time is up
                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());

            // Big explosion in the center
            setTimeout(() => {
                confetti({
                    particleCount: 150,
                    spread: 70,
                    origin: { y: 0.6 }
                });
            }, 500);
        }
